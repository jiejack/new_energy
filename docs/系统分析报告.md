### 5.2 缓存优化方案

#### 缓存策略

| 数据类型 | 缓存时间 | 缓存策略 | 更新机制 |
|---------|---------|---------|---------|
| 区域列表 | 30分钟 | 全量缓存 | 定时刷新 |

#### 缓存预热

```go
func (s *CacheService) Warmup(ctx context.Context) error {
    // 预热区域数据
    regions, _ := s.regionRepo.GetAll(ctx)
    for _, region := range regions {
        s.cache.Set(ctx, fmt.Sprintf("region:%s", region.ID), region, 30*time.Minute)
    }
    
    // 预热电站数据
    stations, _ := s.stationRepo.GetAll(ctx)
    for _, station := range stations {
        s.cache.Set(ctx, fmt.Sprintf("station:%s", station.ID), station, 10*time.Minute)
    }
    
    return nil
}
```

#### 分页优化

```go
// 使用游标分页替代偏移分页
func (r *Repository) ListWithCursor(ctx context.Context, cursor string, limit int) ([]*Entity, string, error) {
    query := r.db.WithContext(ctx).Limit(limit + 1)
    if cursor != "" {
        query = query.Where("id > ?", cursor)
    }
    var entities []*Entity
    if err := query.Find(&entities).Error; err != nil {
        return nil, "", err
    }
    var nextCursor string
    if len(entities) > limit {
        nextCursor = entities[limit].ID
        entities = entities[:limit]
    }
    return entities, nextCursor, nil
}
```
