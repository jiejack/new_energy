```sql
CREATE TABLE IF NOT EXISTS prediction_results (
    id BIGSERIAL PRIMARY KEY,
    prediction_id VARCHAR(64) NOT NULL UNIQUE,
    request_id VARCHAR(64) NOT NULL,
    model_id VARCHAR(128) NOT NULL,
    model_version VARCHAR(32) NOT NULL,
    station_id VARCHAR(64) NOT NULL,
    device_id VARCHAR(64),
    point_id VARCHAR(64),
    
    prediction_type VARCHAR(32) NOT NULL,
    prediction_scope VARCHAR(32),
    prediction_time TIMESTAMP NOT NULL,
    target_time TIMESTAMP NOT NULL,

    confidence_upper FLOAT,
    
    actual_value FLOAT,
    error FLOAT,
    error_percentage FLOAT,
    accuracy_score FLOAT,
    is_accurate BOOLEAN,
    
    input_features JSONB,
    explanation JSONB,
    metadata JSONB,
```

### 6. 分区表策略

```sql
-- 预测结果按月分区
CREATE TABLE prediction_results_2026_04 PARTITION OF prediction_results
FOR VALUES FROM ('2026-04-01') TO ('2026-05-01');

CREATE TABLE prediction_results_2026_05 PARTITION OF prediction_results
FOR VALUES FROM ('2026-05-01') TO ('2026-06-01');

-- 创建自动分区管理函数
CREATE OR REPLACE FUNCTION create_prediction_partition()
RETURNS void AS $$
DECLARE
    next_month DATE;
    partition_name TEXT;
    start_date DATE;
    end_date DATE;
BEGIN
    next_month := date_trunc('month', CURRENT_DATE) + INTERVAL '1 month';
    partition_name := 'prediction_results_' || to_char(next_month, 'yyyy_mm');
    start_date := next_month;
    end_date := next_month + INTERVAL '1 month';
    
    EXECUTE format('
        CREATE TABLE IF NOT EXISTS %I PARTITION OF prediction_results
        FOR VALUES FROM (%L) TO (%L)',
        partition_name, start_date, end_date);
END;
$$ LANGUAGE plpgsql;

-- 每月1号自动创建下个月的分区
CREATE EXTENSION IF NOT EXISTS pg_cron;
SELECT cron.schedule('create-prediction-partition', '0 0 1 * *', 'SELECT create_prediction_partition();');
```
